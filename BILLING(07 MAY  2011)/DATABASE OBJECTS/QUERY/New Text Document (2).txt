select  A.IN_REG_NO,B.PAT_NAME,A.BED_TYPE_NO||A.BED_NO AS BED_NO,B.ADMISSION_DATE ,A.YRCODE
  FROM INDOOR_PAT_BED_INFO  A ,IN_DOOR_PAT_INFO_MAIN B WHERE A.IN_REG_NO=B.IN_REG_NO 
  AND A.YRCODE=B.YRCODE AND B.RELEASE_FLAG=0
  AND A.SERIAL_NO=(SELECT MAX(SERIAL_NO) FROM Indoor_pat_bed_info 
   WHERE in_reg_no =A.IN_REG_NO AND YRCODE=A.YRCODE)
  AND UPPER(A.BED_TYPE)='CABIN'
 ORDER BY B.ADMISSION_DATE DESC 




select A.BED_ext_col||A.BED_NO AS BED_NO
  FROM BED_INFO  A WHERE A.OCCUPY_FLAG =0
    AND A.BED_TYPE='Cabin'


select A.BED_ext_col||A.BED_NO AS BED_NO
  FROM BED_INFO  A WHERE  A.BED_ext_col='13'
  AND DOC_DEPARTMENT<>'COMMON'

select  A.BED_TYPE,A.BED_NO AS BED_NO
  FROM BED_INFO  A WHERE 
   A.BED_ext_col='12' AND  A.DOC_DEPARTMENT<>'COMMON'



select  A.IN_REG_NO,B.PAT_NAME,A.BED_NO AS BED_NO,B.ADMISSION_DATE ,A.YRCODE
  FROM BED_INFO  A ,IN_DOOR_PAT_INFO_MAIN B WHERE A.IN_REG_NO=B.IN_REG_NO 
  AND A.YRCODE=B.YRCODE AND B.RELEASE_FLAG=0
  AND A.SERIAL_NO=(SELECT MAX(SERIAL_NO) FROM Indoor_pat_bed_info 
   WHERE in_reg_no =A.IN_REG_NO AND YRCODE=A.YRCODE)
  AND A.BED_ext_col)'12'
 ORDER BY B.ADMISSION_DATE DESC
