create or replace procedure Rpt_Next_Increment_Report(p_test_cursor in out
test_package.test_type,
P_BeginDate Date,
P_EndDate Date)
as
begin
open p_test_cursor for
SELECT EMP_INFO.EMP_ID,EMP_INFO.EMP_NM, 
ST_DESIG.DESIGNATION,ST_DEPT.DEPT_NM,EMP_INFO.DOB, 
INCREMENT_RECORD.LAST_DT_INCRE,EMP_JOB_INFO.JDATE,EMP_JOB_INFO.BASIC_SAL,
INCREMENT_RECORD.NEXT_DT_INCRE,INCREMENT_RECORD.AMOUNT,INCREMENT_RECORD.UPDATE_DATE
FROM EMP_INFO,EMP_JOB_INFO, 
INCREMENT_RECORD,ST_DEPT,ST_DESIG
WHERE ((EMP_JOB_INFO.EMP_ID=EMP_INFO.EMP_ID)
AND (INCREMENT_RECORD.EMP_ID=EMP_INFO.EMP_ID)
AND (EMP_JOB_INFO.DESIG=ST_DESIG.DESIG_CODE)
AND (EMP_JOB_INFO.DEPT=ST_DEPT.DEPT_CODE)
AND INCREMENT_RECORD.next_DT_INCRE between P_BeginDate and P_EndDate);
end Rpt_Next_Increment_Report;
