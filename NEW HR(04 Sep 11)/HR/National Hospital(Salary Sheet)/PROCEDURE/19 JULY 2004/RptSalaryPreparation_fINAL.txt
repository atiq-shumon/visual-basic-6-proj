create or replace procedure Rpt_Salry_Preparation_Final(p_test_cursor in out
test_package.test_type,
P_Month varchar2,
p_Year varchar2,
p_clASS varchar2)
as
begin
open p_test_cursor for
SELECT SALARY_PREPARATION.EMP_ID, SALARY_PREPARATION.EMP_NM,
SALARY_PREPARATION.DESIGNATION, SALARY_PREPARATION.DEPT_NM, 
SALARY_PREPARATION.ATTN, SALARY_PREPARATION.LEAVE,SALARY_PREPARATION.WORKING_DAY,SALARY_PREPARATION.BASIC, 
SALARY_PREPARATION.H_RENT,SALARY_PREPARATION.MED,SALARY_PREPARATION.CONV, 
SALARY_PREPARATION.TFN,SALARY_PREPARATION.DA,SALARY_PREPARATION.REMARKS, 
SALARY_PREPARATION.ARREAR,SALARY_PREPARATION.TELEPHONE,SALARY_PREPARATION.OTHERS_ALLOWANCE, 
SALARY_PREPARATION.OTHERS_ADDITION,SALARY_PREPARATION.PF_CONTRI_DEDUCTION, 
SALARY_PREPARATION.PF_LN_AMOUNT, 
SALARY_PREPARATION.SALARY_ADVANCE,SALARY_PREPARATION.R_STAMP,SALARY_PREPARATION.OTHERS_DEDUCTION, 
SALARY_PREPARATION.NET_PAYABLE
FROM SALARY_PREPARATION 
where SALARY_PREPARATION.PAY_MONTH=P_Month 
AND SALARY_PREPARATION.PAY_YEAR=p_Year 
AND substr(SALARY_PREPARATION.EMP_ID,1,1)=p_clASS;
end Rpt_Salry_Preparation_Final;
