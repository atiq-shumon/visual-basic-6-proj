SALARY_PREPARATION.ARREAR,SALARY_PREPARATION.TELEPHONE, 
SALARY_PREPARATION.OTHERS_ALLOWANCE,SALARY_PREPARATION.OTHERS_ADDITION, 
SALARY_PREPARATION.PF_CONTRI_DEDUCTION, 
SALARY_PREPARATION.PF_LN_AMOUNT,SALARY_PREPARATION.SALARY_ADVANCE, 
SALARY_PREPARATION.R_STAMP,SALARY_PREPARATION.OTHERS_DEDUCTION, 
SALARY_PREPARATION.EMP_NM,SALARY_PREPARATION.PAY_MONTH, 
SALARY_PREPARATION.PAY_YEAR,SALARY_PREPARATION.DESIGNATION, 
SALARY_PREPARATION.DEPT_NM,SALARY_PREPARATION.BASIC, 
SALARY_PREPARATION.H_RENT, SALARY_PREPARATION.MED, 
SALARY_PREPARATION.CONV,SALARY_PREPARATION.TFN,EMP_JOB_INFO.SCALE_CODE, 
INCREMENT_RECORD.NEXT_DT_INCRE

EMP_JOB_INFO,INCREMENT_RECORD

AND (SALARY_PREPARATION.EMP_ID=EMP_JOB_INFO.EMP_ID))
