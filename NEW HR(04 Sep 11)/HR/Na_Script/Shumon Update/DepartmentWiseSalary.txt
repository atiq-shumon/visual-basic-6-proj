SELECT upper(DEPT_NM),COUNT(DISTINCT EMP_ID) AS DISBURSEE,SUM(BASIC),SUM(H_RENT),SUM(MED),SUM(CONV),SUM(TFN),
SUM(DA),SUM(ARREAR),SUM(TELEPHONE) AS FB,SUM(OTHERS_ALLOWANCE),SUM(OTHERS_ADDITION),
-SUM(PF_CONTRI_DEDUCTION),-SUM(PF_LN_AMOUNT),-SUM(SALARY_ADVANCE),-SUM(R_STAMP),-SUM(OTHERS_DEDUCTION)
-SUM(ADDDEDUCT_OTHER)
  FROM SALARY_PREPARATION
 GROUP BY DEPT_NM
