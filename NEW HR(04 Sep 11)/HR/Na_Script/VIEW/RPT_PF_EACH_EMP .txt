CREATE OR REPLACE  PROCEDURE RPT_PF_EACH_EMP 
(p_test_cursor in out
test_package.test_type,
P_Emp_Id VARCHAR2
)
as
begin
open p_test_cursor for

SELECT 
EP.EMP_ID,
EI.EMP_NM,
SD.DESIGNATION,
D.DEPT_NM,
EP.MONTH,
EP.YEAR,
EP.EMPLOYER_CONTRIBUTION,
EP.EMPLOYEE_CONTRIBUTION,
(EP.EMPLOYER_CONTRIBUTION + EP.EMPLOYEE_CONTRIBUTION) AS TOTA_CON 
FROM 
EMP_PF EP,
EMP_INFO EI,
ST_DESIG SD,
ST_DEPT D,
EMP_JOB_INFO EJ
WHERE 
EP.EMP_ID=EI.EMP_ID
AND EI.EMP_ID=EJ.EMP_ID
AND EJ.DEPT=D.DEPT_CODE
AND EJ.DESIG=SD.DESIG_CODE
AND EP.EMP_ID=P_Emp_Id;
end RPT_PF_EACH_EMP;
