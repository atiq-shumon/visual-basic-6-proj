CREATE OR REPLACE VIEW VIEW_OPENING_CLOSING 
AS 
SELECT 
VTC.EMP_ID,
VTC.EMP_NM,
VTC.DESIGNATION,
VTC.DEPT_NM,
VTC.TOTAL_OFFI_CON,
VTC.TOTAL_EMP_CON,
VTC.TOTAL_CON,
(VTC.TOTAL_CON * 10/100) +(VTC.TOTAL_EMP_CON * 5/100) AS TOTAL_INT, 
VTC.TOTAL_CON + (VTC.TOTAL_CON * 10/100) +(VTC.TOTAL_EMP_CON * 5/100) AS OPENING,
VTC.TOTAL_CON + (VTC.TOTAL_CON * 10/100) +(VTC.TOTAL_EMP_CON * 5/100) AS CLOSING 
from 
VIEW_TOTAL_CONTRIBUTION VTC
