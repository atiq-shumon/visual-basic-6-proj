
--===RPT_GRA_OPEN_CLOSE_ENDOFYEAR

CREATE OR REPLACE  PROCEDURE RPT_GRA_OPEN_CLOSE_ENDOFYEAR
(p_test_cursor in out
test_package.test_type
)
as
begin
open p_test_cursor for

SELECT 
DATE_CLOSING,
OPENING_BALANCE,
CURRENT_YEAR_RECEIVE,
CURRENT_YEAR_PAYMENT,
CLOSING_BALANCE 
FROM 
GRA_OPENING_CLOSING;
END RPT_GRA_OPEN_CLOSE_ENDOFYEAR;

/
